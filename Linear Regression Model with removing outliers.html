    def get_outliers(self,x):
        x=np.array(x.sort_values())
        med=np.median(x)
        med=len(x)//2
        q1=x[:med]
        q3=x[med+1:]
        med_1=q1[len(q1)//2]
        med_3=q3[len(q3)//2]
        iqr=med_3-med_1

        out_2=x[x>(med_3+(1.5*iqr))]
